```python
    # –≠—Ç–∞–ø 4: –ü–∞—Ä–∞–ª–ª–µ–ª—å–Ω—ã–π –ø–æ–∏—Å–∫ –¥–æ–∫—É–º–µ–Ω—Ç–æ–≤ –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –º–∞—Ç–µ—Ä–∏–∞–ª–∞
    # (–±—É–¥–µ—Ç –∑–∞–ø—É—â–µ–Ω–æ –ø–æ—Å–ª–µ —Ç–æ–≥–æ, –∫–∞–∫ –ê–û–°–† –≥–æ—Ç–æ–≤—ã)
    def start_parallel_search(aosr_results):
        materials = extract_materials(aosr_results)
        # –ó–∞–ø—É—Å–∫–∞–µ–º –ø–∞—Ä–∞–ª–ª–µ–ª—å–Ω–æ –¥–ª—è –∫–∞–∂–¥–æ–≥–æ –º–∞—Ç–µ—Ä–∏–∞–ª–∞
        search_tasks = group(
            search_document_agent_task.s(material)
            for material in materials
        )
        return search_tasks.apply_async()

    # –≠—Ç–∞–ø 5: –ü—Ä–æ–≤–µ—Ä–∫–∞ –∫–æ–º–ø–ª–µ–∫—Ç–Ω–æ—Å—Ç–∏
    validation_task = validation_agent_task.si(project_id)

    # –≠—Ç–∞–ø 6: –§–æ—Ä–º–∏—Ä–æ–≤–∞–Ω–∏–µ —Ñ–∏–Ω–∞–ª—å–Ω–æ–≥–æ –ø–∞–∫–µ—Ç–∞
    package_task = package_creation_task.si(project_id)

    # –°–æ–±–∏—Ä–∞–µ–º –≤ —Ü–µ–ø–æ—á–∫—É (chain)
    pipeline = chain(
        ocr_task,
        analysis_task,
        aosr_task,
        # –ü–æ—Å–ª–µ –ê–û–°–† ‚Üí –ø–∞—Ä–∞–ª–ª–µ–ª—å–Ω—ã–π –ø–æ–∏—Å–∫ –¥–æ–∫—É–º–µ–Ω—Ç–æ–≤ ‚Üí –ø—Ä–æ–≤–µ—Ä–∫–∞ ‚Üí –ø–∞–∫–µ—Ç
        chord(
            header=start_parallel_search,
            body=validation_task
        ),
        package_task
    )

    # –ó–∞–ø—É—Å–∫–∞–µ–º
    result = pipeline.apply_async()
    return result.id
```
